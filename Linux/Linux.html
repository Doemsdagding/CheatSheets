(un)install

Description

Command
install package
 sudo pacman -S <package_name>
search if package exists
 pacman -Ss <search>
uninstall package
 sudo pacman -R <package_name>
remove cache of uninstalled packages
 sudo pacman -Sc
remove orphaned packages
 sudo pacman -Rns $(pacman -Qdtq)
install AUR package
 yay -S <package_name>
uninstall AUR package
 yay -Rns <package_name>
sear for AUR package
 yay -Ss <keyword>
